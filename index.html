Combining Dependent p-values

2026-06-29

Supplementary R code for reproducing the examples in the paper ``Combining Dependent p-values: Methods and Properties from a Practical Perspective’’ by Gasparin M., Goeman J.J., Heller R. and Solari A.

Domain-Level Analysis of the Functional Observational Battery

The functional observational battery (FOB) is a biological screening assay consisting of \(m=25\) endpoints grouped into six domains, each reflecting different aspects of neurological function, and is used to assess neurophysiological effects in animals following toxin exposure.

Domain-wise combined \(p\)-values.

# ---- Domain-wise combination 
domains <- endpoints_by_domain[-length(endpoints_by_domain)]
domains$All <- 1:25
domains <- c(tail(domains, 1), head(domains, -1))

comb_funs <- c("Bonferroni","Simes","Hommel+","Fisher","Average","Median","Cauchy","Harmonic")

res <- matrix(NA, nrow = length(domains), ncol = length(comb_funs) * 2)
colnames(res) <- c(comb_funs, paste0("U-",comb_funs))
rownames(res) <- paste(names(domains), " (",sapply(domains,length),")", sep="")

for (d in seq_along(domains)) {
  
  Zsel <- Z[, domains[[d]]]
  Tstat <- matrix(NA, nrow = B, ncol = ncol(Zsel))
  
  for (b in 1:B) {
    Zperm <- rbind(Zsel[perms[, b], ], Zsel[-perms[, b], ])
    Tstat[b, ] <- sapply(
      1:ncol(Zperm),
      function(i) Tmax(Zperm[1:n, i], Zperm[(n + 1):N, i])
    )
  }
  
  pval <- apply(Tstat, 2, function(t) mean(t >= t[1]))
  
  Pmat <- apply(Tstat, 2, function(t) {
    sapply(seq_along(t), function(i) mean(t >= t[i]))
  })
  
  # ---- Joint permutation p-values ----
  
  minP <- apply(Pmat, 1, min)
  res[d,1] <- mean(-minP >= -minP[1])
  
  SimesP <- apply(Pmat, 1, function(x) min(sort(x) * length(x)/(1:length(x))))
  res[d,2] <- mean(-SimesP >= -SimesP[1])
  
  HommelP <- apply(Pmat, 1, function(x) min(sort(x) * length(x)/(1:length(x))))
  res[d,3] <- mean(-HommelP >= -HommelP[1])
  
  FisherP <- apply(Pmat, 1, function(x) -sum(log(x)))
  res[d,4] <- mean(FisherP >= FisherP[1])
  
  AverageP <- rowMeans(Pmat)
  res[d,5] <- mean(-AverageP >= -AverageP[1])
  
  MedianP <- apply(Pmat, 1, function(x) sort(x)[ceiling(length(x)/2)])
  res[d,6] <- mean(-MedianP >= -MedianP[1])
  
  CauchyP <- apply(Pmat, 1, function(x) atan(mean(tan(pi*(x-1/2))) )/pi+1/2 )
  res[d,7] <- mean(-CauchyP >= -CauchyP[1])
  
  HarmonicP <- apply(Pmat, 1, function(x) mean(x^(-1))^(1/(-1)) )
  res[d,8] <- mean(-HarmonicP >= -HarmonicP[1])
  
  # ---- Marginal combination ----
  
  res[d,9]  <- min(1, min(pval) * length(pval))
  res[d,10] <- min(1, min(sort(pval) * length(pval)/(1:length(pval))))
  res[d,11] <- pSimes(pval, method = "G")
  res[d,12] <- pmean(pval, r = 0, dependence = "A")
  res[d,13] <- min(1, 2 * mean(pval))
  res[d,14] <- min(1, 2 * sort(pval)[ceiling(length(pval)/2)])
  res[d,15] <- NA
  res[d,16] <- ifelse(length(pval)>2, pharmonic(pval, method = "H2", censor = TRUE), NA )
}

Simulation study

We consider the Gaussian location-shift model with equicorrelated covariance matrix. For each combination of sparsity level \(k\) and correlation \(\rho\), the signal strength \(h\) is calibrated so that Simes’ test at level \(\alpha=0.05\) attains target power \(1-\beta = 0.7\). The following code computes the calibration matrix:

source("find_h_Simes.R")

m <- 30
eps <- 1e-4
l <- 10

rhoVec <- seq(-1/(m - 1) + eps, 1 - eps, length.out = l)
sVec <- round(seq(1, m, length.out = min(m, l)))

nreps_h <- 1000
alpha <- 0.05
target_power <- 0.7
tol_power <- 0.001

set.seed(123)

h_mat <- matrix(
  NA_real_,
  length(sVec), length(rhoVec),
  dimnames = list(paste0("s=", sVec), paste0("rho=", round(rhoVec, 2)))
)

for (i in seq_along(rhoVec)) {
  rho <- rhoVec[i]
  Sigma <- matrix(rho, m, m); diag(Sigma) <- 1
  R <- chol(Sigma)

  for (j in seq_along(sVec)) {
    h_mat[j, i] <- find_h_for_target_power_vec(
      s = sVec[j], R = R, m = m,
      target_power = target_power,
      nreps_h = nreps_h,
      tol_power = tol_power,
      alpha = alpha
    )
  }
}

We next compare the empirical power of the combining methods under the Gaussian location-shift model using the calibrated values of \(h\):

nreps <- 1000

methods <- c("Average", "Bonferroni", "Cauchy", "Fisher",
             "Harmonic", "Hommel+", "Median", "Stouffer")

comb <- array(NA, c(nreps, length(sVec), length(rhoVec), length(methods)))
dimnames(comb)[[4]] <- methods

for (i in seq_along(rhoVec)) {
  rho <- rhoVec[i]
  Sigma <- matrix(rho, m, m); diag(Sigma) <- 1
  R <- chol(Sigma)

  for (j in seq_along(sVec)) {
    s <- sVec[j]
    theta <- numeric(m); theta[1:s] <- h_mat[j, i] / s

    X <- theta + t(R) %*% matrix(rnorm(m * nreps), m, nreps)
    P <- pnorm(X, lower.tail = FALSE)

    comb[, j, i, "Average"]    <- 2 * colMeans(P) < alpha
    comb[, j, i, "Bonferroni"] <- m * apply(P, 2, min) < alpha
    comb[, j, i, "Median"]     <- 2 * apply(P, 2, median) < alpha
    comb[, j, i, "Fisher"]     <- apply(P, 2, \(p) pmean(p, 0, "A", TRUE) < alpha)
    comb[, j, i, "Harmonic"]   <- apply(P, 2, \(p) pharmonic(p, "H2") < alpha)
    comb[, j, i, "Cauchy"]     <- apply(P, 2, \(p) pCauchy(p, "A", alpha)$Results[2] == "Reject")

    comb[, j, i, "Hommel+"] <- apply(P, 2, \(p) {
      H <- sum(1 / seq_len(m))
      sum((H * p <= alpha) / ceiling(m * H * p / alpha)) >= 1
    })

    comb[, j, i, "Stouffer"] <- colSums(qnorm(1 - P)) >= m * qnorm(1 - alpha)
  }
}

i_sel <- unique(round(c(1, length(rhoVec)/3, 2*length(rhoVec)/3, length(rhoVec))))
j_sel <- unique(round(c(1, length(sVec)/3, 2*length(sVec)/3, length(sVec))))

df_i <- map_dfr(i_sel, \(i)
